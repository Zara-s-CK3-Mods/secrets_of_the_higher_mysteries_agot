		}
	}
}

zz_valyria_can_become_apprentice = {
	OR = {
		not = {
			has_trait = zz_magister 
			has_trait = dragon 
		}
		
		
		any_spouse = {
			not = {
				has_trait = zz_magister
				has_trait = dragon 
			}
		}
		any_close_or_extended_family_memebr = {
			not = { 
				has_trait = zz_magister
				has_trait = dragon 
			}
		}
		house ?= { 
			any_house_member = {
				not = { 
					has_trait = zz_magister
					has_trait = dragon 
				}
			}
		}
		any_courtier_or_guest = { 
			not = { 
				has_trait = zz_magister
				has_trait = dragon 
			}
		}
		any_vassal_or_below = { 
			not = { 
				has_trait = zz_magister
				has_trait = dragon 
			}
		}
	}
}

	OR = {
		zz_valyria_is_mage = yes 
		zz_valyria_is_scroll_holder = yes
	}
}

zz_valyria_is_scroll_holder = {
	OR = { 
		any_character_artifact = {
			has_variable = spell
		}
		any_courtier_or_guest = {
			any_character_artifact = {
				has_variable = spell
			}
		}
	}
}
zz_valyria_is_mage_can_teach = {
	OR = { 
		AND = {
			has_trait = zz_magister

			}
		}
		any_living_mage = {
			AND = {
				liege_or_court_owner ?= PREV
				NOT = {

				}
			}
		}
	}
}
