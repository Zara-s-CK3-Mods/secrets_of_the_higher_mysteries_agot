# Duel Success values

zz_magic_duel_success_medium = {
	value = 0 
	if = {
		limit = {
			has_variable = magic_type
		}
		add = {
			switch = {
				trigger ?= var:magic_type
				flag:dream = {
					value = zzz_valyria_magic_dream_magic_08_value
					multiply = 1.75
				}
				flag:blood = {
					value = zz_valyria_magic_blood_magic_17_value
					multiply = 1.2
				}
				flag:alchemy = { 
					value = zz_valyria_magic_alchemy_magic_17_value
					multiply = 1
				}
				flag:shadow = { 
					value = zz_valyria_magic_shadow_magic_09_value
					multiply = 1
				}
				flag:fire = {
					value = zz_valyria_magic_fire_magic_01_value
					multiply = 1.5
				}
				flag:water = {
					value = zz_valyria_magic_water_magic_01_value
					multiply = 1.25
				}
				flag:air = {
					value = zz_valyria_magic_air_magic_01_value
					multiply = 2
				}
			}
		}
	}
	
	min = 50
}
